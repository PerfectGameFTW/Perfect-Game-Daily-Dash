const squareClient = new Client({
  accessToken: process.env.SQUARE_ACCESS_TOKEN || '',
  environment: Environment.Production
});

function convertToEasternTime(utcTimestamp: string): string {
  const utcDate = new Date(utcTimestamp);
  const estDate = toZonedTime(utcDate, 'America/New_York');
  return formatInTimeZone(estDate, 'America/New_York', 'yyyy-MM-dd HH:mm:ss');
}

async function syncPayments() {
  const response = await squareClient.paymentsApi.listPayments();
  const payments = response.result.payments || [];
  for (const payment of payments) {
    await db.insert(transactions).values({
      squareId: payment.id,
      amount: payment.amountMoney.amount / 100, // Convert cents to dollars
      categoryId: 'retail', // Placeholder, refine with logic
      status: payment.status,
      timestamp: convertToEasternTime(payment.createdAt),
      squareData: payment
    }).onConflictDoNothing();
  }
}
