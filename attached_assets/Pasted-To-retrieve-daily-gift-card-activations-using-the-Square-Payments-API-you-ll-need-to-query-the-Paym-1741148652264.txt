To retrieve daily gift card activations using the Square Payments API, you’ll need to query the Payments endpoint and filter transactions where the gift_card payment type is used. Here’s how you can do it:

1. Define Your Date Range (Eastern Time)
Since you’re working with business data, ensuring your date range aligns with Eastern Time is crucial. You already have a getEasternDateRange function in dateUtils.js, which you can use to get the correct UTC range.

javascript
Copy
Edit
const { getEasternDateRange } = require('./dateUtils');
2. Call the Square Payments API
You'll use the SearchOrders API or ListPayments API to find payments that were gift card activations.

Using ListPayments API
javascript
Copy
Edit
const { client } = require('./squareClient'); // Ensure you have Square client setup
const { getEasternDateRange } = require('./dateUtils');

async function getGiftCardActivations(dateRange = 'today') {
    const { start, end } = getEasternDateRange(dateRange);
    
    try {
        const { result } = await client.paymentsApi.listPayments({
            beginTime: start.toISOString(),
            endTime: end.toISOString(),
            sortOrder: 'ASC',
        });

        // Filter only gift card activations (capture method: 'GIFT_CARD_ACTIVATION')
        const giftCardActivations = result.payments.filter(payment => 
            payment.cardDetails?.card?.cardType === 'GIFT_CARD'
        );

        console.log('Gift Card Activations:', giftCardActivations);
        return giftCardActivations;
    } catch (error) {
        console.error('Error fetching gift card activations:', error);
    }
}

// Example usage
getGiftCardActivations('today');
Alternative: Using SearchOrders API
If you’re using Square Orders, you can filter transactions by GIFT_CARD category:

javascript
Copy
Edit
async function searchGiftCardActivations(dateRange = 'today') {
    const { start, end } = getEasternDateRange(dateRange);

    try {
        const { result } = await client.ordersApi.searchOrders({
            locationIds: ['YOUR_LOCATION_ID'], // Replace with your Square location ID
            query: {
                filter: {
                    dateTimeFilter: {
                        createdAt: {
                            startAt: start.toISOString(),
                            endAt: end.toISOString(),
                        },
                    },
                    fulfillmentFilter: {
                        fulfillmentTypes: ['GIFT_CARD'],
                    },
                },
            },
        });

        console.log('Gift Card Orders:', result.orders);
        return result.orders;
    } catch (error) {
        console.error('Error searching gift card activations:', error);
    }
}

// Example usage
searchGiftCardActivations('today');
3. Ensure API Permissions
Make sure your Square application has the following permissions:

PAYMENTS_READ
ORDERS_READ
GIFT_CARDS_READ
